# 변수와 메서드를 사용해서 코드를 보다 깔끔하고 효율적으로 개선하는 방법

## Step1

```shell
package practice02;

public class VariableAndMethod {

	public static void main(String[] args) {

		printA();
                printA();
		printA();
		printA();

		printB();
		printB();
		printB();
		printB();

	}

	public static void printA() {

		System.out.println("----");
		System.out.println("A");
		System.out.println("A");

	}

	public static void printB() {

		System.out.println("****");
		System.out.println("B");
		System.out.println("B");

	}
}
```
## Step2

```shell

package practice02;

public class VariableAndMethod {

	public static void main(String[] args) {

		printA("----");
		printA("----");
		printA("----");
		printA("----");

		printB("****");
		printB("****");
		printB("****");
		printB("****");

	}

	public static void printA(String delimiter) {

		System.out.println(delimiter);
		System.out.println("A");
		System.out.println("A");

	}

	public static void printB(String delimiter) {

		System.out.println(delimiter);
		System.out.println("B");
		System.out.println("B");

	}
}
```

## Step3

```shell
package practice02;

public class VariableAndMethod {

	public static void main(String[] args) {
		String delimiter = "----";
		
		printA(delimiter);
		printA(delimiter);
		printA(delimiter);
		printA(delimiter);
		
		delimiter = "****";
		
		printB(delimiter);
		printB(delimiter);
		printB(delimiter);
		printB(delimiter);
		
	}

	public static void printA(String delimiter) {
		
		System.out.println(delimiter);
		System.out.println("A");
		System.out.println("A");
		
	}
	
	public static void printB(String delimiter) {
		
		System.out.println(delimiter);
		System.out.println("B");
		System.out.println("B");
		
	}
}
```
